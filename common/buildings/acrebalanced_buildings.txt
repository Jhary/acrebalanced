		category = planet_buildings
		cost = {

			energy = 100
		}			
		upkeep = {      
			energy = 4

#	}

	ai_weight = {
		weight = 5
	}
  
  triggered_planet_modifier = {
		potential = {
	  	exists = owner
      owner = { is_gestalt = no }
		}
		modifier = {
      job_administrator_add = 2
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_gestalt = yes }					
		}
		modifier = {
      job_coordinator_add = 3
		}
	}
	
	triggered_desc = {
		trigger = {

		}
		text = job_administrator_effect_desc
	}
	
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		text = job_coordinator_effect_desc
	}	
}
